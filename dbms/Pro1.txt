Objective: To learn how to:

==> Step 1: Create a new database
CREATE DATABASE MCA;

-- Step 2: Use the database
USE MCA;

-- Step 3: Create the 'account' table
CREATE TABLE account (
    acc_no INT PRIMARY KEY,
    amount DECIMAL(10, 2)
);
